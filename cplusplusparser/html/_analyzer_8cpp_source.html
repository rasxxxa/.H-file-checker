167std::unordered_map<std::string, VariableOccurrenceReadWrite> Analyzer::CheckVariablesForUsage(const std::set<std::string>& variables, const std::vector<std::string>& sourceFile)
168{
169 std::unordered_map<std::string, VariableOccurrenceReadWrite> variableOccurrenceReadWrite;
170
171 for (const auto& variable : variables)
172 {
173 VariableOccurrenceReadWrite occurrenceReadWrite{};
174 int num_of_writes = 0;
175 int num_of_reads = 0;
176 int num_of_function_calls = 0;
177 for (const auto& line : sourceFile)
178 {
179 if (line.contains(variable))
180 {
181 auto varPos = line.find(variable);
182
183 if (line.find('=') < varPos
184 || (line.find('(') < varPos && line.find(')', varPos) != std::string::npos)
185 || line.find("return") != std::string::npos
186 || (line.find('[') < varPos && line.find(']', varPos) > varPos)
187 || line.find("==") != std::string::npos
188 || line.find("!=") != std::string::npos)
189 num_of_reads++;
190 else if (line.find('=', varPos) != std::string::npos
191 || line.find("++", varPos) != std::string::npos
192 || line.find("--", varPos) != std::string::npos)
193 num_of_writes++;
194 else if (line.find("->", varPos) != std::string::npos
195 || line.find(variable + '.', varPos) != std::string::npos)
196 num_of_function_calls++;
197
198 }
199 }
200 occurrenceReadWrite.write_times = num_of_writes;
201 occurrenceReadWrite.read_times = num_of_reads;
202 occurrenceReadWrite.method_times = num_of_function_calls;
203 variableOccurrenceReadWrite[variable] = occurrenceReadWrite;
204 }
